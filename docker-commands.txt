# ── Redis (short-term memory) ─────────────────────────────────────────────────
docker run -d `
  --name my-redis `
  -p 6379:6379 `
  redis:latest


# ── Analytics DB (restaurant business tables) ─────────────────────────────────
#
# Database : zenvyra
# Port     : 5433
# User     : admin_user (owner), ca_agent_user (read-only, used by the SQL agent)
#
# init.sql creates:
#   - All restaurant business tables (orders, menu_items, customers, etc.)
#   - ca_agent_user with read-only access (pg_read_all_data)
#
# NOTE: PostgreSQL 18+ requires mount at /var/lib/postgresql (not /data)
# NOTE: docker-entrypoint-initdb.d only runs on first start (empty data dir)

docker run -d `
  --name analytics-db `
  -e POSTGRES_USER=admin_user `
  -e POSTGRES_PASSWORD=admin_password `
  -e POSTGRES_DB=zenvyra `
  -p 5433:5432 `
  -v C:\MyData\personal-learning\analytics-data:/var/lib/postgresql `
  -v C:\MyData\personal-learning\pgscript\init.sql:/docker-entrypoint-initdb.d/01_init.sql `
  pgvector/pgvector:pg18


# ── Memory DB (long-term memory + audit logs) ─────────────────────────────────
#
# Database : zenvyra_memory
# Port     : 5434
# User     : admin_user (owner, used by the app for read/write)
#
# init_memory.sql creates:
#   - pgvector extension (required for LangGraph AsyncPostgresStore)
#
# 02_memory.sql creates:
#   - memory schema
#   - memory.query_log    (audit log per conversation)
#   - memory.agent_steps  (audit log per ReAct step)
#   - Views over public.store (LangGraph conversation summaries)
#
# LangGraph AsyncPostgresStore.setup() creates automatically:
#   - public.store        (conversation summary documents)
#   - public.store_vectors (pgvector embeddings)

docker run -d `
  --name memory-db `
  -e POSTGRES_USER=admin_user `
  -e POSTGRES_PASSWORD=admin_password `
  -e POSTGRES_DB=ca_memory `
  -p 5434:5432 `
  -v C:\MyData\personal-learning\memory-data:/var/lib/postgresql `
  -v C:\MyData\personal-learning\pgscript\init_memory.sql:/docker-entrypoint-initdb.d/01_init_memory.sql `
  pgvector/pgvector:pg18


# ── Useful commands ───────────────────────────────────────────────────────────

# Connect to analytics DB
docker exec -it analytics-db psql -U admin_user -d zenvyra

# Connect to memory DB
docker exec -it memory-db psql -U admin_user -d ca_memory

# Run memory schema manually on existing memory-db container
docker exec -it memory-db psql -U admin_user -d ca_memory -f /docker-entrypoint-initdb.d/01_init_memory.sql

# Stop and remove all containers
docker stop my-redis analytics-db memory-db
docker rm my-redis analytics-db memory-db
